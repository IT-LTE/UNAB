import java.util.ArrayList;
import java.util.Scanner;

public class GestorClientes {
    private ArrayList<Cliente> clientes = new ArrayList<>();
    private Scanner sc = new Scanner(System.in);
    private ClienteInput input = new ClienteInput(sc);

    public void ejecutar() {
        int opcion;

        do {
            mostrarMenu();
            opcion = leerOpcion();

            switch (opcion) {
                case 1:
                    Cliente nuevoCliente = input.leerCliente();
                    clientes.add(nuevoCliente);
                    System.out.println("‚úÖ Cliente agregado correctamente.\n");
                    break;
                case 2:
                    mostrarClientes();
                    break;
                case 3:
                    System.out.println("üëã Saliendo del sistema...");
                    break;
                default:
                    System.out.println("‚ùå Opci√≥n inv√°lida. Intente nuevamente.\n");
            }

        } while (opcion != 3);
    }

    private void mostrarMenu() {
        System.out.println(" ");
        System.out.println("====== Compa√±ia de seguro ======");
        System.out.println(" ");
        System.out.println("==== Men√∫ Principal ====");
        System.out.println("1. Ingresar nuevo cliente");
        System.out.println("2. Mostrar todos los clientes");
        System.out.println("3. Salir");
        System.out.print("Opci√≥n: ");
    }

    private int leerOpcion() {
        String entrada = sc.nextLine().trim();
        if (entrada.matches("\\d+")) {
            return Integer.parseInt(entrada);
        }
        return -1; // Valor inv√°lido
    }

    private void mostrarClientes() {
        if (clientes.isEmpty()) {
            System.out.println("No hay clientes ingresados.\n");
            return;
        }

        System.out.println("Clientes registrados:");
        for (int i = 0; i < clientes.size(); i++) {
            System.out.println("Cliente #" + (i + 1));
            clientes.get(i).mostrarCliente();
            System.out.println("------------------------");
        }
    }
}
